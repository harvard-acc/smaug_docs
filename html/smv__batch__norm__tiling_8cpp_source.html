39 void TilingOptimizer::enumPostFCTilingConfigs(
40  TensorShape inputsShape,
41  TensorShape weightsShape,
42  int maxTileSize,
43  std::array<TilingDims, 2> strategies,
44  std::list<TilingConfig>& fullConfigs) {
45  TilingDims inputTilingDims = strategies[0];
46  TilingDims weightTilingDims = strategies[1];
47  // Supported tiling dims: None, DimN and DimNC for inputs. None and DimNC
48  // for weights.
49  // The tiling config enumeration goes as follows:
50  // 1. Start with inputs. Enumerate all shapes that fit.
51  // 2. Move on to weights and outputs. Enumerate all shapes that are
52  // compatible with the input shape and fit. The outputs tile use the same
53  // tile shape as the inputs.
54  // 3. For all tiling strategy, compute the total SRAM utilization. The
55  // highest one is the chosen one.
56  assert(inputTilingDims == None || inputTilingDims == DimN ||
57  inputTilingDims == DimNC);
58  assert(weightTilingDims == None || weightTilingDims == DimNC);
59  std::vector<TensorShape> inputsConfigs;
60  if (inputTilingDims == DimN) {
61  std::vector<int> minShape = inputsShape.dims();
62  minShape[0] = 1;
63  enum2DTensorTilingConfigs(
64  inputsShape, maxTileSize, minShape, { 1, 1 }, inputsConfigs);
65  } else if (inputTilingDims == DimNC) {
66  enum2DTensorTilingConfigs(inputsShape,
67  maxTileSize,
68  { 1, kVectorSize },
69  { 1, kVectorSize },
70  inputsConfigs);
71  } else {
72  inputsConfigs.push_back(inputsShape);
73  }
74  assert(!inputsConfigs.empty() && "No tiling configurations found!");
75 
76  // Fill in weights and outputs.
77  for (auto it = inputsConfigs.begin(); it != inputsConfigs.end(); ++it) {
78  TensorShape& inputsConfig = *it;
79  if (weightTilingDims == DimNC) {
80  if (needsCwiseTiling(inputTilingDims)) {
81  // If the inputs are also tiled activation-wise, then the
82  // weights have to take the same activations dimension.
83  TilingConfig config;
84  config.weights = weightsShape;
85  config.weights[1] = inputsConfig[1];
86  if (config.weights.storageSize() <= maxTileSize) {
87  config.inputs = inputsConfig;
88  config.outputs = inputsConfig;
89  fullConfigs.push_back(config);
90  } else {
91  break;
92  }
93  } else {
94  int minChannels = std::min(weightsShape[1], kVectorSize);
95  for (int c = minChannels; c <= weightsShape[1];
96  c += kVectorSize) {
97  TilingConfig config;
98  config.weights = weightsShape;
99  config.weights[1] = c;
100  if (config.weights.storageSize() <= maxTileSize) {
101  config.inputs = inputsConfig;
102  config.outputs = inputsConfig;
103  fullConfigs.push_back(config);
104  } else {
105  break;
106  }
107  }
108  }
109  } else {
110  TilingConfig config(inputsConfig, weightsShape, inputsConfig);
111  fullConfigs.push_back(config);
112  }
113  }
114  assert(!fullConfigs.empty() && "No tiling configurations found!");
115 }

125  // Supported tiling dims: DimN, DimNC, DimNH, DimNW, DimNHW, DimNCH and
126  // DimNCW for inputs. None for weights for now. For a 32KB weights spad, it
127  // would mean the weights have more than 4096 channels if tiling is
128  // required.
129  // TODO: add other tiling dims for weights if we need that later.

133  assert(inputTilingDims == None || inputTilingDims == DimN ||
134  inputTilingDims == DimNC || inputTilingDims == DimNH ||
135  inputTilingDims == DimNW || inputTilingDims == DimNHW ||
136  inputTilingDims == DimNCH || inputTilingDims == DimNCW);
137  assert(weightTilingDims == None);
138  std::vector<TensorShape> inputsConfigs;
139  if (inputTilingDims == DimN) {
140  std::vector<int> minShape = inputsShape.dims();
141  minShape[0] = 1;
142  enum4DTensorTilingConfigs(inputsShape,
143  maxTileSize,
144  minShape,
145  { 1, 1, 1, 1 },
146  inputsConfigs);
147  } else if (inputTilingDims == DimNC) {
148  std::vector<int> minShape = inputsShape.dims();
149  minShape[0] = 1;
150  minShape[3] = kVectorSize;
151  enum4DTensorTilingConfigs(inputsShape,
152  maxTileSize,
153  minShape,
154  { 1, 1, 1, kVectorSize },
155  inputsConfigs);
156  } else if (inputTilingDims == DimNH) {
157  std::vector<int> minShape = inputsShape.dims();
158  minShape[0] = 1;
159  minShape[1] = kVectorSize;
160  enum4DTensorTilingConfigs(inputsShape,
161  maxTileSize,
162  minShape,
163  { 1, kVectorSize, 1, 1 },
164  inputsConfigs);
165  } else if (inputTilingDims == DimNW) {
166  std::vector<int> minShape = inputsShape.dims();
167  minShape[0] = 1;
168  minShape[2] = kVectorSize;
169  enum4DTensorTilingConfigs(inputsShape,
170  maxTileSize,
171  minShape,
172  { 1, 1, kVectorSize, 1 },
173  inputsConfigs);
174  } else if (inputTilingDims == DimNHW) {
175  std::vector<int> minShape = { 1, kVectorSize, kVectorSize,
176  inputsShape[3] };
177  enum4DTensorTilingConfigs(inputsShape,
178  maxTileSize,
179  minShape,
180  { 1, kVectorSize, kVectorSize, 1 },
181  inputsConfigs);
182  } else if (inputTilingDims == DimNCH) {
183  std::vector<int> minShape = { 1, kVectorSize, inputsShape[2],
184  kVectorSize };
185  enum4DTensorTilingConfigs(inputsShape,
186  maxTileSize,
187  minShape,
188  { 1, kVectorSize, 1, kVectorSize },
189  inputsConfigs);
190  } else if (inputTilingDims == DimNCW) {
191  std::vector<int> minShape = { 1, inputsShape[1], kVectorSize,
192  kVectorSize };
193  enum4DTensorTilingConfigs(inputsShape,
194  maxTileSize,
195  minShape,
196  { 1, 1, kVectorSize, kVectorSize },
197  inputsConfigs);
198  } else {
199  inputsConfigs.push_back(inputsShape);
200  }
201  assert(!inputsConfigs.empty() && "No tiling configurations found!");
202 
203  // Fill in weights and outputs.
204  for (auto it = inputsConfigs.begin(); it != inputsConfigs.end(); ++it) {
205  TilingConfig config(*it, weightsShape, *it);
206  fullConfigs.push_back(config);
207  }

211 // Determine the best basic tiling shape for this batch norm layer.
212 //
213 // The algorithm first determines the dimensions along which the inputs,
214 // weights, and outputs will be tiled. Then based on those dimensions, we
215 // enumerate all possible basic tile shapes for inputs, weights, and outputs. A
216 // **basic** shape is the shape that all but potentially the last tile along a
217 // set of dimensions will use. This triplet of tile shapes defines a
218 // TilingConfig. The TilingConfig that maximizes the total combined size of
219 // input, weights, and output tiles is chosen as the best.
220 //
221 // This algorithm assumes that the maximum tile size for weights, inputs, and
222 // outputs are all the same and that they will reside in separate scratchpads
223 // (no sharing).
224 //
225 // Args:
226 // inputs: Inputs tensor of the batch norm operator.
227 // weights: A tensor that concatenates the four weights tensors of the batch
228 // norm operator.
229 // outputs: Outputs tensor of the batch norm operator.
230 // Returns:
231 // The TilingConfig that describes the best tiling shapes.
232 //
233 TilingConfig TilingOptimizer::computeBasicTileShapes(Tensor* inputs,
234  Tensor* weights,
235  Tensor* outputs) {
236  int maxTileSize = SmvBackend::SpadSize() / inputs->getDataTypeSize();
237  // The outputs have the same shape as the inputs. No need to tile it.
238  assert(inputs->getShape() == outputs->getShape());
239  std::array<TilingDims, 2> strategies =
240  determineBestTilingDims(inputs, weights, maxTileSize);
241  TilingDims inputTilingDims = strategies[0];
242  TilingDims weightTilingDims = strategies[1];
243  TilingDims outputTilingDims = inputTilingDims;
244 
245  dout(2) << " Tiling dimensions chosen: \n"
246  << " input: " << inputTilingDims
247  << ", weight: " << weightTilingDims
248  << ", output: " << inputTilingDims << "\n";
249 
250  TensorShape inputsShape = inputs->getShape();
251  TensorShape weightsShape = weights->getShape();
252  std::list<TilingConfig> fullConfigs;
253  bool isPostConv = (inputs->ndims() == 4);
254  if (isPostConv) {
255  enumPostConvTilingConfigs(inputsShape,
256  weightsShape,
257  maxTileSize,
258  strategies,
259  fullConfigs);
260  } else {
261  enumPostFCTilingConfigs(inputsShape,
262  weightsShape,
263  maxTileSize,
264  strategies,
265  fullConfigs);
266  }
267 
268  dout(2) << " Number of possible tiling configs: " << fullConfigs.size()
269  << "\n";
270  for (auto& config : fullConfigs)
271  dout(2) << " " << config << "\n";
272  auto maxIt = std::max_element(
273  fullConfigs.begin(),
274  fullConfigs.end(),
275  [](const TilingConfig& c1, const TilingConfig& c2) {
276  return c1.getTotalSize() < c2.getTotalSize();
277  });
278  assert(maxIt != fullConfigs.end() && "Failed to get best tiling config!");
279  // Fill in the tiling dims.
280  (*maxIt).inputTilingDims = inputTilingDims;
281  (*maxIt).weightTilingDims = weightTilingDims;
282  (*maxIt).outputTilingDims = outputTilingDims;
283  return *maxIt;
284 }

286 std::array<TiledTensor, 3> TilingOptimizer::doTiling(SmvBatchNormOp* op) {
287  auto inputs = op->getInput(SmvBatchNormOp::Inputs);
288  auto mean = op->getInput(SmvBatchNormOp::Mean);
289  auto variance = op->getInput(SmvBatchNormOp::Variance);
290  auto gamma = op->getInput(SmvBatchNormOp::Gamma);
291  auto beta = op->getInput(SmvBatchNormOp::Beta);
292  // Concatenate the four weight tensors into one.
293  auto weights = concatTensors(
294  { mean, variance, gamma, beta }, 0, op->getWorkspace());
295  auto outputs = op->getOutput(SmvBatchNormOp::Outputs);
296  TilingConfig tileConfig =
297  TilingOptimizer::computeBasicTileShapes(inputs, weights, outputs);
298  TiledTensor tiledInputs =
299  generateTiledTensor(inputs, tileConfig.inputs, op);
300  // Copy data for the weight tiles since the data is read-only.
301  TiledTensor tiledWeights =
302  generateTiledTensorAndCopyData(weights, tileConfig.weights, op);
303  TiledTensor tiledOutputs =
304  generateTiledTensor(outputs, tileConfig.inputs, op);
305  return { tiledInputs, tiledWeights, tiledOutputs };
306 }
