47  TensorShape inferWeightsShape() const {
48  TensorShape shape = getInput(Inputs)->getShape();
49  DataLayout layout = shape.getLayout();
50  int ndims = shape.ndims();
51  if (ndims >= 4) {
52  // This is a volume which should be batch norm'ed by feature map.
53  bool isNCHW = layout == DataLayout::NCHW;
54  int fmaps = isNCHW ? shape[ndims - 3] : shape[ndims - 1];
55  return TensorShape(
56  { 1, fmaps }, DataLayout::NC, Backend::Alignment);
57  } else if (ndims == 2) {
58  if (layout == DataLayout::NC)
59  return TensorShape(
60  { 1, shape[1] }, DataLayout::NC, Backend::Alignment);
61  else
62  assert(false && "Unexpected data layout for batch norm!");
63  } else {
64  assert(false && "Unexpected input dimensions for batch norm!");
