15 std::pair<activation_type, activation_param_t> getActivationParams(
16  UnaryOp<SmvBackend>* op) {
17  activation_type function;
18  activation_param_t params;
19  OpType opType = op->getOpType();
20  if (opType == OpType::ReLU) {
21  auto reluOp = dynamic_cast<SmvReluOp*>(op);
22  params.slope = reluOp->getSlope();
23  if (params.slope > 0)
24  function = activation_type::LRELU;
25  else
26  function = activation_type::RELU;
27  } else if (opType == OpType::ELU) {
28  function = activation_type::ELU;
29  auto eluOp = dynamic_cast<SmvEluOp*>(op);
30  params.alpha = eluOp->getAlpha();
31  } else if (opType == OpType::SELU) {
32  function = activation_type::SELU;
33  auto seluOp = dynamic_cast<SmvSeluOp*>(op);
34  params.alpha = seluOp->getAlpha();
35  params.lambda = seluOp->getLambda();
36  } else if (opType == OpType::Tanh) {
37  function = activation_type::TANH;
38  } else if (opType == OpType::HardTanh) {
39  function = activation_type::HARD_TANH;
40  auto hardTanhOp = dynamic_cast<SmvHardTanhOp*>(op);
41  params.min = hardTanhOp->getMin();
42  params.max = hardTanhOp->getMax();
43  } else if (opType == OpType::Sigmoid) {
44  function = activation_type::SIGMOID;
45  } else if (opType == OpType::Softmax) {
46  assert(false && "Softmax should call its own run() implementation!");
47  }
48  return { function, params };
49 }

79 TiledTensor generateTiles(Tensor* tensor,
80  const TensorShape& tileShape,
81  Operator* op,
82  bool copyData) {
83  const TensorShape& inputShape = tensor->getShape();
84  int inputSize = inputShape.storageSize();
85  int tileSize = tileShape.storageSize();
86  int numTiles = std::ceil(inputSize * 1.0 / tileSize);
87  TiledTensor tiledTensor(
88  TensorShape({ 1, numTiles }, DataLayout::NC), tensor, true);
89  int remainingSize = inputSize;
90  int srcOffset = 0;
91  for (auto tileIndex = tiledTensor.startIndex(); !tileIndex.end();
92  ++tileIndex) {
93  int currentTileSize = std::min(remainingSize, tileSize);
94  TensorShape currentShape({ 1, currentTileSize },
95  DataLayout::NC,
96  tileShape.getAlignment());
97  std::string tileName = op->getName() + ":" + tensor->getName() +
98  "/tile:" + std::to_string((int)tileIndex);
99  Tensor* tile = new Tensor(tileName, currentShape);
100  tile->allocateStorage(tensor->getDataType());
101  tiledTensor.setTile(tileIndex, { srcOffset }, tile, copyData);
102  srcOffset += currentTileSize;
103  remainingSize -= currentTileSize;
104  }
105  op->getWorkspace()->addTiledTensor(tiledTensor);
106  dout(1) << " Tiled Tensor " << tensor->getName() << ":\n"
107  << " original tensor shape: " << tensor->getShape() << "\n"
108  << " tile shape " << tileShape
109  << ", number of tiles: " << tiledTensor.size() << "\n";
110  return tiledTensor;
111 }

113 std::array<TiledTensor, 2> doTiling(UnaryOp<SmvBackend>* op, bool copyData) {
114  auto inputs = op->getInput(UnaryOp<SmvBackend>::Inputs);
115  auto outputs = op->getOutput(UnaryOp<SmvBackend>::Outputs);
116  // The tiling for unary operators can be greatly simplified in comparison to
117  // other operators. The tile shape is determined as [1, spadSize].
118  int maxTileSize =
119  std::min(SmvBackend::SpadSize() / inputs->getDataTypeSize(),
120  inputs->getShape().storageSize());
121  TensorShape tileShape(
122  { 1, maxTileSize }, DataLayout::NC, SmvBackend::Alignment);
123  TiledTensor tiledInputs = generateTiles(inputs, tileShape, op, copyData);
124  TiledTensor tiledOutputs = generateTiles(outputs, tileShape, op, copyData);
125  return { tiledInputs, tiledOutputs };
126 }
