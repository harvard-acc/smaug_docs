13 // Determine the best tiling dimensions for running inner product on SMV.
14 //
15 // This function imposes some additional constraints on the tiling dimensions,
16 // in that certain combinations of input/weight/output tiling dimensions are
17 // not allowed in the interest of tiling code complexity.
18 //
19 // Returns:
20 // A 3-element array of TilingDims enums (inputs, weights, outputs).
21 std::array<TilingDims, 3> TilingOptimizer::determineBestTilingDims(
22  Tensor* inputs, Tensor* weights, Tensor* outputs, int maxTileSize) {
23  // Determine the best tiling strategy for each of inputs, weights, and
24  // outputs. Don't try to figure out the actual tile sizes yet.
25  TilingDims bestInputTilingDims = findBestTilingDims(
26  inputs->getShape(), maxTileSize, { 1, kNumMaccsPerPE });
27  TilingDims bestWeightTilingDims = findBestTilingDims(
28  weights->getShape(), maxTileSize, { kNumPEs, kNumMaccsPerPE });
29  TilingDims bestOutputTilingDims = findBestTilingDims(
30  outputs->getShape(), maxTileSize, { 1, kNumPEs });
31 
32  // Apply some constraints to simplify tiling logic.
33  //
34  // If weights require tiling on neurons, then outputs must be DimNC (if
35  // outputs require tiling), so that we will copy out C neurons of outputs
36  // after every tile.
37  if (needsNwiseTiling(bestWeightTilingDims) && bestOutputTilingDims != None)
38  bestOutputTilingDims = DimNC;
39 
40  return { bestInputTilingDims, bestWeightTilingDims, bestOutputTilingDims };
41 }

43 // Determine the best basic tiling shape for this fc layer without bias.
44 //
45 // The algorithm first determines the dimensions along which the inputs,
46 // weights, and outputs will be tiled. Then based on those dimensions, we
47 // enumerate all possible basic tile shapes for inputs, weights, and outputs. A
48 // **basic** shape is the shape that all but potentially the last tile along a
49 // set of dimensions will use. This triplet of tile shapes defines a
50 // TilingConfig. The TilingConfig that maximizes the total combined size of
51 // input, weights, and output tiles is chosen as the best.
52 //
53 // To limit the number of possibilities, we only enumerate each dimension in
54 // certain increments. For example, input channels are only enumerated in
55 // multiples of kNumMaccsPerPE, and output channels are only enumerated in
56 // multiples in kNumPEs.
57 //
58 // This algorithm assumes that the maximum tile size for weights, inputs, and
59 // outputs are all the same and that they will reside in separate scratchpads
60 // (no sharing).
61 //
62 // Args:
63 // op: The SMV inner product operator. All tensors must have been created
64 // with createAllTensors() prior to calling this function.
65 // Returns:
66 // The TilingConfig that describes the best tiling shapes.
67 //
68 TilingConfig TilingOptimizer::computeBasicTileShapes(SmvInnerProductOp* op) {
69  Tensor* inputs = op->getInput(op->Inputs);
70  Tensor* weights = op->getInput(op->Weights);
71  Tensor* outputs = op->getOutput(op->Outputs);
72  int maxTileSize = SmvBackend::SpadSize() / inputs->getDataTypeSize();
73  std::array<TilingDims, 3> strategies =
74  determineBestTilingDims(inputs, weights, outputs, maxTileSize);
75  TilingDims inputTilingDims = strategies[0];
76  TilingDims weightTilingDims = strategies[1];
77  TilingDims outputTilingDims = strategies[2];
78 
79  dout(1) << " Tiling dimensions chosen:\n"
80  << " input: " << inputTilingDims
81  << ", weight: " << weightTilingDims
82  << ", output: " << outputTilingDims << "\n";
83 
84  TensorShape inputsShape = inputs->getShape();
85  TensorShape weightsShape = weights->getShape();
86  TensorShape outputsShape = outputs->getShape();
87 
88  // There are two degrees of freedom we can play with in total:
89  // N (batch/neuron), C (activation).
90  // Each tiling strategy may reduce this down to just three.
91  // 1. Start with inputs. Enumerate all shapes that fit.
92  // 2. Move on to weights. Enumerate all shapes that are compatible with
93  // the input shape and fit.
94  // 3. Move on to outputs. If the weights don't need tiling, the outputs
95  // can be tiled independently; otherwise, based on the input and weights
96  // tile shapes, the output tile shape is completely determined.
97  // For all tiling strategy, compute the total SRAM utilization. The highest
98  // one is the chosen one.
99  std::vector<TensorShape> inputConfigs;
100  if (inputTilingDims == DimN) {
101  enum2DTensorTilingConfigs(inputsShape,
102  maxTileSize,
103  { 1, inputsShape[1] },
104  { 1, 1 },
105  inputConfigs);
106  } else if (inputTilingDims == DimNC) {
107  std::vector<int> minShape = inputsShape.dims();
108  enum2DTensorTilingConfigs(inputsShape,
109  maxTileSize,
110  { 1, kNumMaccsPerPE },
111  { 1, kNumMaccsPerPE },
112  inputConfigs);
113 
114  } else {
115  inputConfigs.push_back(inputsShape);
116  }
117  assert(!inputConfigs.empty() && "No tiling configurations found!");
118 
119  // Fill in weights.
120  std::list<TilingConfig> inputWeightConfigs;
121  for (auto it = inputConfigs.begin(); it != inputConfigs.end(); ++it) {
122  const TensorShape& inputsShape = *it;
123  if (weightTilingDims == DimN) {
124  int minOfmaps = std::min(weightsShape[0], kNumPEs);
125  for (int n = minOfmaps; n <= weightsShape[0]; n += kNumPEs) {
126  TilingConfig config;
127  config.weights = TensorShape({ n, inputsShape[1] },
128  inputsShape.getLayout(),
129  SmvBackend::Alignment);
130  if (config.weights.storageSize() <= maxTileSize) {
131  config.inputs = inputsShape;
132  inputWeightConfigs.push_back(config);
133  } else {
134  break;
135  }
136  }
137  } else if (weightTilingDims == DimNC) {
138  int minNeurons = std::min(weightsShape[0], kNumPEs);
139  int minActs = std::min(weightsShape[1], kNumMaccsPerPE);
140  for (int n = minNeurons; n <= weightsShape[0]; n += kNumPEs) {
141  TilingConfig config;
142  config.weights = weightsShape;
143  config.weights[0] = n;
144  if (needsCwiseTiling(inputTilingDims)) {
145  // If the inputs are also tiled activation-wise, then the
146  // weights have to take the same activations dimension.
147  config.weights[1] = inputsShape[1];
148  if (config.weights.storageSize() <= maxTileSize) {
149  config.inputs = inputsShape;
150  inputWeightConfigs.push_back(config);
151  } else {
152  break;
153  }
154  } else {
155  // The weights can be independently tiled activation-wise
156  // only if the inputs are not tiled on activations.
157  for (int c = minActs; c <= weightsShape[1];
158  c += kNumMaccsPerPE) {
159  config.weights[1] = c;
160  if (config.weights.storageSize() <= maxTileSize) {
161  config.inputs = inputsShape;
162  inputWeightConfigs.push_back(config);
163  } else {
164  break;
165  }
166  }
167  }
168  }
169  } else {
170  TilingConfig config;
171  config.inputs = inputsShape;
172  config.weights = weightsShape;
173  if (needsCwiseTiling(inputTilingDims)) {
174  // This can happen with small weights. If the inputs are tiled
175  // channelwise, then the weight tile need to have the same
176  // number of channels.
177  config.weights[1] = inputsShape[1];
178  }
179  inputWeightConfigs.push_back(config);
180  }
181  }
182  assert(!inputWeightConfigs.empty() && "No tiling configurations found!");
183 
184  // Fill in outputs.
185  std::vector<TilingConfig> fullConfigs;
186  for (auto it = inputWeightConfigs.begin(); it != inputWeightConfigs.end();
187  ++it) {
188  int minChannels = std::min(it->weights[0], kNumPEs);
189  bool weightsNeedTiling = (weightTilingDims != None);
190  bool outputsNeedTiling = (outputTilingDims != None);
191  for (int c = minChannels; c <= weightsShape[0]; c += kNumPEs) {
192  TilingConfig config = *it;
193  config.outputs = outputsShape;
194  config.outputs[0] = config.inputs[0];
195  if (weightsNeedTiling && outputsNeedTiling) {
196  config.outputs[1] = config.weights[0];
197  } else if (outputsNeedTiling) {
198  // This could rarely happen, but for completeness let's keep it.
199  // If the weights don't need tiling and the outputs need tiling,
200  // the channel size of the output tile size can be determined
201  // independently.
202  config.outputs[1] = c;
203  }
204  if (config.outputs.storageSize() <= maxTileSize) {
205  fullConfigs.push_back(config);
206  }
207  // This means the output shape is uniquely determined, so we don't
208  // need to explore any other output channel values.
209  if (weightsNeedTiling || outputTilingDims == None)
210  break;
211  }
212  }
213  dout(2) << " Number of possible tiling configs: " << fullConfigs.size()
214  << "\n";
215  for (auto& config : fullConfigs)
216  dout(2) << " " << config << "\n";
217  auto maxIt = std::max_element(
218  fullConfigs.begin(),
219  fullConfigs.end(),
220  [](const TilingConfig& c1, const TilingConfig& c2) {
221  return c1.getTotalSize() < c2.getTotalSize();
222  });
223  assert(maxIt != fullConfigs.end() && "Failed to get best tiling config!");
224  // Fill in the tiling dims.
225  maxIt->inputTilingDims = inputTilingDims;
226  maxIt->weightTilingDims = weightTilingDims;
227  maxIt->outputTilingDims = outputTilingDims;
228  return *maxIt;
229 }

231 std::array<TiledTensor, 3> TilingOptimizer::doTiling(SmvInnerProductOp* op) {
232  auto input = op->getInput(SmvInnerProductOp::Inputs);
233  auto kernels = op->getInput(SmvInnerProductOp::Weights);
234  auto output = op->getOutput(SmvInnerProductOp::Outputs);
235  TilingConfig tileConfig = TilingOptimizer::computeBasicTileShapes(op);
236  TiledTensor tiledInputs = generateTiledTensor(input, tileConfig.inputs, op);
237  // Copy data for the weight tiles since the data is read-only.
238  TiledTensor tiledWeights =
239  generateTiledTensorAndCopyData(kernels, tileConfig.weights, op);
240  TiledTensor tiledOutputs =
241  generateTiledTensor(output, tileConfig.outputs, op);
242  return { tiledInputs, tiledWeights, tiledOutputs };
243 }
