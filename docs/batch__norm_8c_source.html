13 // 1/sqrt(var + eps) is precomputed to avoid having to run a sqrt and division
14 // in the ASIC.
15 ALWAYS_INLINE
16 v8fp_t batch_norm_simd_op(v8fp_t input,
17  v8fp_t mean,
18  v8fp_t recip_sqrt_var,
19  v8fp_t gamma,
20  v8fp_t beta) {
21  v8fp_t scale = recip_sqrt_var * gamma;
22  v8fp_t shift = input - mean;
23  return shift * scale + beta;
24 }
