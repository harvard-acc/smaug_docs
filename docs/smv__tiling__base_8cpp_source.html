10 // Find the best set of dimensions to tile a given tensor shape.
11 //
12 // The goal is to divide up a tensor into tiles that each are <= maxTileSize
13 // elements. The tensor layouts can be NC, NHWC or NCHW. The minimum tile size
14 // is specified via minShape. The preferences for tiling dimensions are as
15 // follows:
16 // 1) No tiling.
17 // 2) Dim-N tiling. N is the input batch.
18 // 3) Dim-NC tiling. After tiling by N, tile channelwise. Do not tile in HW.
19 // 4) Dim-NH tiling. After tiling by N, tile rowwise. Do not tile in WC.
20 // 5) Dim-NW tiling. After tiling by N, tile columnwise. Do not tile in HC.
21 // 6) Dim-NHW tiling. After tiling by N, tile rowwise and then columnwise. Do
22 // not tile in C.
23 // 7) Dim-NCH tiling. After tiling by N and channel dimensions, tile rowwise.
24 // Do not tile in W.
25 // 8) Dim-NCW tiling. After tiling by N and channel dimensions, tile
26 // columnwise. Do not tile in H.
27 //
28 // Except Option 1, a minimum size for each dimension can be deduced from
29 // minShape. Note that for a 2D tensor shape, only options 1-5 are viable.
30 TilingDims TilingOptimizerBase::findBestTilingDims(
31  const TensorShape& shape,
32  int maxTileSize,
33  const std::vector<int>& minShape) {
34  DataLayout layout = shape.getLayout();
35  assert(layout == NHWC || layout == NCHW || layout == NC);
36  if (shape.storageSize() <= maxTileSize)
37  return TilingDims::None;
38  int minN = std::min(shape[0], minShape[0]);
39  bool isNHWC = layout == NHWC;
40  int cIdx = isNHWC ? 3 : 1;
41  int minC = std::min(shape[cIdx], minShape[cIdx]);
42  int sizePerN = shape.storageSize() / shape[0];
43  if (sizePerN * minN <= maxTileSize)
44  return TilingDims::DimN;
45  if (sizePerN * (minC * 1.0 / shape[cIdx]) <= maxTileSize)
46  return TilingDims::DimNC;
47  if (shape.ndims() == 2) {
48  std::cerr << "[ERROR]: Unable to find a supported set of tiling "
49  "dimensions for 2D tensor with shape "
50  << shape << "!\n";
51  exit(1);
52  }
53  int hIdx = isNHWC ? 1 : 2;
54  int wIdx = isNHWC ? 2 : 3;
55  int minH = std::min(shape[hIdx], minShape[hIdx]);
56  int minW = std::min(shape[wIdx], minShape[wIdx]);
57  if (sizePerN * (minH * 1.0 / shape[hIdx]) <= maxTileSize)
58  return TilingDims::DimNH;
59  if (sizePerN * (minW * 1.0 / shape[wIdx]) <= maxTileSize)
60  return TilingDims::DimNW;
61  if (sizePerN * (minH * 1.0 / shape[hIdx]) * (minW * 1.0 / shape[wIdx]) <=
62  maxTileSize)
63  return TilingDims::DimNHW;
64  if (sizePerN * (minC * 1.0 / shape[cIdx]) * (minH * 1.0 / shape[hIdx]) <=
65  maxTileSize)
66  return TilingDims::DimNCH;
67  if (sizePerN * (minC * 1.0 / shape[cIdx]) * (minW * 1.0 / shape[wIdx]) <=
68  maxTileSize)
69  return TilingDims::DimNCW;
70  std::cerr << "[ERROR]: Unable to find a supported set of tiling dimensions "
71  "for 4D tensor with shape "
72  << shape << "!\n";
73  exit(1);
74 }

76 void TilingOptimizerBase::enum2DTensorTilingConfigs(
77  TensorShape shape,
78  int maxTileSize,
79  const std::vector<int>& minShape,
80  const std::vector<int>& strides,
81  std::vector<TensorShape>& configs) {
82  int minN = std::min(minShape[0], shape[0]);
83  int minC = std::min(minShape[1], shape[1]);
84  int strideN = strides[0];
85  int strideC = strides[1];
86  for (int n = minN; n <= shape[0]; n += strideN) {
87  for (int c = minC; c <= shape[1]; c += strideC) {
88  TensorShape config(
89  { n, c }, shape.getLayout(), shape.getAlignment());
90  if (config.storageSize() <= maxTileSize)
91  configs.push_back(config);
92  else
93  break;
94  }
95  }

98 void TilingOptimizerBase::enum4DTensorTilingConfigs(
99  TensorShape shape,
100  int maxTileSize,
101  const std::vector<int>& minShape,
102  const std::vector<int>& strides,
103  std::vector<TensorShape>& configs) {
104  bool isNHWC = shape.getLayout() == NHWC;
105  int idxH = isNHWC ? 1 : 2;
106  int idxW = isNHWC ? 2 : 3;
107  int idxC = isNHWC ? 3 : 1;
108  int minN = std::min(minShape[0], shape[0]);
109  int minH = std::min(minShape[idxH], shape[idxH]);
110  int minW = std::min(minShape[idxW], shape[idxW]);
111  int minC = std::min(minShape[idxC], shape[idxC]);
112  int strideN = strides[0];
113  int strideH = strides[idxH];
114  int strideW = strides[idxW];
115  int strideC = strides[idxC];
116  for (int n = minN; n <= shape[0]; n += strideN) {
117  for (int c = minC; c <= shape[idxC]; c += strideC) {
118  for (int h = minH; h <= shape[idxH]; h += strideH) {
119  for (int w = minW; w <= shape[idxW]; w += strideW) {
120  TensorShape config;
121  if (isNHWC) {
122  config = TensorShape({ n, h, w, c },
123  shape.getLayout(),
124  shape.getAlignment());
125  } else {
126  config = TensorShape({ n, c, h, w },
127  shape.getLayout(),
128  shape.getAlignment());
129  }
130  if (config.storageSize() <= maxTileSize)
131  configs.push_back(config);
132  else
133  break;
134  }
135  }
136  }
137  }
