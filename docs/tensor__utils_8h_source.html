31 void writeTensorToOstream(std::ostream& os, const Tensor& tensor) {
32  const TensorShape& shape = tensor.getShape();
33  if (shape.ndims() == 0) {
34  os << " [ ]\n";
35  return;
36  }
37  int ndims = shape.ndims();
38  int newlineAfterElems = shape[ndims - 1];
39  int newGroupAfterElems =
40  (shape.ndims() >= 2 ? shape[ndims - 1] * shape[ndims - 2]
41  : shape[ndims - 1]);
42  int counter = 0;
43  const DType* data = tensor.template data<DType>();
44  os << tensor.getName() << ", shape = " << shape << "\n";
45  for (auto idx = tensor.startIndex(); !idx.end(); ++idx) {
46  // Print the current index after going through all of the last two
47  // dimensions.
48  if (counter == 0)
49  os << idx << "\n[ ";
50  printTensorElement<DType>(os, data, idx);
51  os << " ";
52  ++counter;
53  if (counter % newGroupAfterElems == 0) {
54  counter = 0;
55  os << " ]\n";
56  } else if (counter % newlineAfterElems == 0) {
57  os << "\n ";
58  }
59  }
60 }

64 template <typename DType>
65 void copyTensorRegion(Tensor* dest,
66  Tensor* src,
67  const std::vector<int>& destOrigin,
68  const std::vector<int>& srcOrigin,
69  const std::vector<int>& regionSize) {
70  const TensorShape& srcShape = src->getShape();
71  const TensorShape& destShape = dest->getShape();
72  TensorShape regionShape(
73  regionSize, srcShape.getLayout(), srcShape.getAlignment());
74  const int ndims = srcShape.ndims();
75  auto destIt = TensorRegionIndexIterator(destShape, destOrigin, regionSize);
76  auto srcIt = TensorRegionIndexIterator(srcShape, srcOrigin, regionSize);
77 
78  // We know where to copy data from and how much data we should copy (the
79  // data region), now starting from the last dimension, we figure out how
80  // much contiguous data there exists such that we can apply more efficient
81  // data copy mechanisms (memcpy).
82  std::vector<int> contiguousRegion(ndims, 1);
83  int contiguousSize = 1;
84  for (int i = ndims - 1; i >= 0; i--) {
85  contiguousSize *= regionShape.getStorageDim(i);
86  contiguousRegion[i] = regionShape[i];
87  // If we find a region dimension smaller than that of either src or
88  // dest tensor, then the next region dimension must not be contiguous.
89  if (regionShape[i] < srcShape[i] || regionShape[i] < destShape[i])
90  break;
91  }
92 
93  // Copy the data region from the src tensor to the dest tensor.
94  DType* destPtr = dest->template data<DType>();
95  DType* srcPtr = src->template data<DType>();
96  while (!srcIt.end() && !destIt.end()) {
97 #ifdef PEDANTIC
98  destPtr[destIt] = srcPtr[srcIt];
99  ++destIt;
100  ++srcIt;
101 #else
102  memcpy(&destPtr[destIt],
103  &srcPtr[srcIt],
104  contiguousSize * sizeof(DType));
105  destIt += contiguousRegion;
106  srcIt += contiguousRegion;
107 #endif
108  }
109 }
